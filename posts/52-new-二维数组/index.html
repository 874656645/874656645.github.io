new 二维数组

int main( )
{  
    int ** pArr = NULL;
    pArr = new int*[4];
    for(int i = 0; i < 4; i++){
        pArr[i] = new int[8];
    }
    for(int i = 0; i < 4; ++i){
        for(int j = 0; j < 8; ++j){
            pArr[i][j] = i * j;
        }
    }
    for(int i = 0; i < 4; ++i){
        for(int j = 0; j < 8; ++j){
            if(j == 0) cout << endl;
            cout << pArr[i][j] << '\t';
        }
    }
    // 删除
    for(int i = 0; i < 4; ++i){
        delete [] pArr[i];
    }
    delete [] pArr;
    
    return 0;
}
