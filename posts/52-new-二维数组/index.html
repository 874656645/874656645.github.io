int main( )
{
    int a[3];           // 全部初始化为 0
    int b[3]{4};        // 第一个值是 4，其余初始化为 0
    int c[3] = {4};     // 同上

    // 与 vector 不同
    vector<int> v(3, 4);// 3 个值都是 4
    return 0;
}
