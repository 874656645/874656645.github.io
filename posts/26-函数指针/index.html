函数指针

December 16, 2021 · 1 min · Rick Cui 
#include <iostream>
#include <cstdlib>
using namespace std;

// 声明函数指针变量
int (*fn)(int) = NULL;

// 定义函数指针类型
typedef int (*Fn)(int);

int cal(int v){ 
    return v;
};

// 函数指针做为形参
void proc(int v, Fn f){
    cout << f(v) << endl;
};
int main()
{
    fn = cal;
    cout << (*fn)(10) << endl;
    cout << fn(10) << endl;
    
    Fn f1 = cal;
    cout << (*f1)(20) << endl;

    proc(30, cal);
    
    return 0;
}
