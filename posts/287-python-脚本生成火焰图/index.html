Python 脚本生成火焰图

December 15, 2024 · 3 min · Rick Cui

  1. 首先下载脚本代码。

  2. 安装 ActivePerl_5.16.2.3010812913.msi 并配置环境变量。

  3. 将 Python 脚本放到 FlameGraph-master 目录下运行。

python 脚本：

import argparse
import subprocess
import sys
import os

print(sys.version)

def execute_cmd(cmd):
    ret = subprocess.run(cmd, capture_output=True, shell=True, encoding='gb2312')
    if ret.returncode == 0:
        print(f'command executed successfully: {cmd}.')
    else:
        print(ret.stderr)
        print(f'command executed failed: {cmd}.')
    return ret

def execute_cmd_redirect_to_file(cmd, file_name):
    with open(file_name, 'w') as f:
        ret = subprocess.run(cmd, text=True, stdout=f, shell=True, encoding='utf-8')
    if ret.returncode == 0:
        print(f'command executed successfully: {cmd}.')
    else:
        print(ret.stderr)
        print(f'command executed failed: {cmd}.')
    return ret

def push_simpleperf_to_device(ndk_path):
    simpleperf_path_on_device = '/data/local/tmp/simpleperf'
    adb_command = f'adb shell ls -l {simpleperf_path_on_device}'
    ret = execute_cmd(adb_command)
    if ret.returncode == 0:
        print(f'{simpleperf_path_on_device} exists')
    else:
        simpleperf_path = f'{ndk_path}/simpleperf/bin/android/arm64/simpleperf'
        if os.path.exists(simpleperf_path):
            adb_command = f'adb push {simpleperf_path} {simpleperf_path_on_device}'
            ret = execute_cmd(adb_command)
            if ret.returncode == 0:
                print(f'{simpleperf_path} pushed to {simpleperf_path_on_device}')
            else:
                print(f'{simpleperf_path} push failed')
                exit()
        else:
            print(f'{simpleperf_path} not exists')
            exit()
# 根据焦点的 app 获取包名和 activity
def get_package_name_and_activity():
    adb_command = 'adb shell "dumpsys window | grep mCurrentFocus"'
    ret = execute_cmd(adb_command)
    print(f'CurrentFocus APP window is {ret.stdout}')
    if ret.returncode != 0:
        print(ret.stderr)
        print('get package name and activity failed.')
        exit()
    package_name_and_activity = ret.stdout.split(' ')[-1]
    package_name = package_name_and_activity.split('/')[0]
    activity_name = package_name_and_activity.split('/')[1][:-2]
    return package_name, activity_name

# 根据 perf.data 生成火焰图
def generate_flame_graph_by_perf_data(ndk_path):
    if not os.path.exists('perf.data'):
        print('perf.data not exists')
        exit()
    
    adb_command = f'python {ndk_path}/simpleperf/report_sample.py'
    file_name = '2.perf'
    ret = execute_cmd_redirect_to_file(adb_command, file_name)
    if ret.returncode == 0:
        print(f'{file_name} generated successfully.')
    else:
        print(ret.stderr)
        print(f'{file_name} generated failed.')
        exit()
    adb_command = 'perl stackcollapse-perf.pl 2.perf'
    file_name = '3.fold'
    ret = execute_cmd_redirect_to_file(adb_command, file_name)
    if ret.returncode == 0:
        print(f'{file_name} generated successfully.')
    else:
        print(ret.stderr)
        print(f'{file_name} generated failed.')
        exit()
    adb_command = 'perl flamegraph.pl 3.fold'
    file_name = '4.svg'
    ret = execute_cmd_redirect_to_file(adb_command, file_name)
    if ret.returncode == 0:
        print(f'{file_name} generated successfully.')
        print('Flame graph generated successfully.')
    else:
        print(ret.stderr)
        print(f'{file_name} generated failed.')
        print('Flame graph generated failed.')
        exit()

def generate_perf_data_by_pid(args):
    # 获取应用的 pid
    adb_command = f'adb shell pidof {args.app}'
    ret = execute_cmd(adb_command)
    if ret.returncode != 0:
        print(ret.stderr)
        print(f'get the pid of {args.app} failed.')
        exit()
    pid = int(ret.stdout)
    print(f'the pid of {args.app} is {pid}')
    script = f'{args.ndk_path}/simpleperf/app_profiler.py'
    adb_command = f'python {script} --pid {pid} -r "-e task-clock:u -f 100 -g --duration {args.duration}" --ndk_path {args.ndk_path}'
    ret = execute_cmd(adb_command)
    if ret.returncode == 0:
        print(f'perf.data generated successfully based on app pid {args.app}.')
    else:
        print(ret.stderr)
        print(f'perf.data generated failed based on app pid {pid}.')

def generate_perf_data_by_app_name(args):
    # 根据当前焦点的 app 获取包名和 activity
    package_name, activity_name = get_package_name_and_activity()
    print(package_name, activity_name)
    script = f'{args.ndk_path}/simpleperf/app_profiler.py'
    adb_command = f'python {script} -p {package_name} -a {activity_name} -r "-e task-clock:u -f 100 -g --duration {args.duration}" --ndk_path {args.ndk_path}'
    ret = execute_cmd(adb_command)
    if ret.returncode == 0:
        print('perf.data generated successfully based on package name and activity.')
    else:
        print(ret.stderr)
        print('perf.data generated failed based on package name and activity.')

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument('-p', '--app', type=str, default = None, help="""Profile an Android app, given the package name.
                              Like `-p com.example.android.myapp`.""")
    parser.add_argument('-d', '--duration', type=int, default = 60, help="""Profile an Android app, given the duration in seconds.
                              Like `-d 180`.""")
    parser.add_argument('-np', '--ndk_path', type=str, default = None,
                             help="""Set the path of a ndk release. app_profiler.py needs some
                                     tools in ndk, like readelf.""")
    args = parser.parse_args()
    print(args.app)
    print(args.duration)
    print(args.ndk_path)

    if args.app is None:
        print('Please input the package name.')
        exit()

    if os.path.exists('perf.data'):
        os.remove('perf.data')
    if os.path.exists('2.perf'):
        os.remove('2.perf')
    if os.path.exists('3.fold'):
        os.remove('3.fold')
    if os.path.exists('4.svg'):
        os.remove('4.svg')
        
    # 尝试 root 并 remount
    adb_root = True
    adb_command = 'adb root'
    ret = execute_cmd(adb_command)
    if ret.stdout.find('adbd cannot run as root') != -1:
        print('adb root failed.')
        adb_root = False
    if adb_root:
        adb_command = 'adb remount'
        ret = execute_cmd(adb_command)

    # 判断 /data/local/tmp/simpleperf 文件是否存在
    # 如果不存在，则将电脑本地的 simpleperf 拷贝到设备上
    push_simpleperf_to_device(args.ndk_path)

    if adb_root:
        generate_perf_data_by_pid(args)
    else:
        generate_perf_data_by_app_name(args)
        
    # 根据 perf.data 生成火焰图
    generate_flame_graph_by_perf_data(args.ndk_path)

注意： 需要 Android Studio 中 simpleperf 的支持，也可将 simpleperf 目录拷贝到 FlameGraph-master 目录，然后进入 simpleperf 文件夹下通过 cmd 命令执行以下步骤：

  • 记录数据 python app_profiler.py -p com.xxx.xxx -a com.xxx.xxx.MainActivity -r "-e task-clock:u -f 100 -g --duration 600"

  • 执行 python report_sample.py >../2.perf 转换数据

  • 进入 FlameGraph-master 文件夹下通过 cmd 命令进入命令行

  • 执行 perl stackcollapse-perf.pl 2.perf >3.fold 转换数据

  • 数据转换完成执行 perl flamegraph.pl 3.fold >4.svg 生成火焰图
