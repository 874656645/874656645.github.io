数据结构与算法——数组中的逆序对

December 12, 2024 · 1 min · Rick Cui

问题链接

要求： 空间复杂度 O(n)，时间复杂度 O(nlogn)

思路： 归并排序，在合并的阶段统计逆序对个数

#include <vector>
class Solution {
    int count{0};
    void mergeSort(vector<int>& nums, int l, int h){
        if(l >= h) return;
        int mid = l + (h - l) / 2;
        mergeSort(nums, l, mid);
        mergeSort(nums, mid+1, h);
        merge(nums, l, mid, h);
    }
    void merge(vector<int>& nums, int l, int mid, int h){
        vector<int> tmp(h - l + 1);
        int s = l;
        int i = 0;
        int j = mid + 1;
        while (l <= mid && j <= h) {
            if(nums[l] <= nums[j]){
                tmp[i] = nums[l];
                l++;
                i++;
            }else{
                tmp[i] = nums[j];
                j++;
                i++;
                // 统计逆序对
                count += mid - l + 1;
                count %= 1000000007;
            }
        }

        while (l <= mid) {
            tmp[i++] = nums[l++];
        }
        while (j <= h) {
            tmp[i++] = nums[j++];
        }

        for (auto it : tmp) {
            nums[s++] = it;
        }
    }
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param nums int整型vector 
     * @return int整型
     */
    int InversePairs(vector<int>& nums) {
        // write code here
        if(nums.size() < 2) return 0;

        mergeSort(nums, 0, nums.size()-1);

        return count;
    }
};
