SDF 生成有向距离场图片

December 15, 2024 · 4 min · Rick Cui

Dead Reckoning算法生成有向距离场

例如原图：

效果：根据距离边界像素的距离与给定的透明像素距离（m_pixelDist）修改圆形内部像素的透明度

C++ 代码：

class SDFTexture
{
public:
	SDFTexture(rs::Texture*, float pixelDist);
	~SDFTexture();

	void generateSDFTex();
private:

	rs::Texture* m_SDFTex;
	int32 m_width, m_height;
	float m_pixelDist;
	std::vector<std::vector<PixelType> > m_pixelMap;
	std::vector<std::vector<float> > m_distanceMap;
	std::vector<std::vector<Point16> > m_positionMap;
};

SDFTexture::SDFTexture(rs::Texture* tex, float pixelDist)
{
	m_SDFTex = tex;
	m_width = m_SDFTex->width();
	m_height = m_SDFTex->height();
	m_pixelDist = pixelDist;
	m_distanceMap = std::vector<std::vector<float> >(m_height + 2, std::vector<float>(m_width + 2, FLT_MAX));
	m_positionMap = std::vector<std::vector<Point16> >(m_height + 2, std::vector<Point16>(m_width + 2, { INT16_MAX, INT16_MAX }));
	m_pixelMap = std::vector<std::vector<PixelType> >(m_height + 2, std::vector<PixelType>(m_width + 2, Pixel_fromA8R8G8B8(0x0)));
}

SDFTexture::~SDFTexture()
{
	m_SDFTex = NULL;
}

void SDFTexture::generateSDFTex()
{
	// 1. 从显存中初始化位图
	unsigned char* picData = m_SDFTex->newPixelDataInCPU();
	for (int32 row = 1; row <= m_height; ++row)
	{
		for (int32 col = 1; col <= m_width; ++col)
		{
			// 当前像素索引
			int32 curPixIndex = ((row - 1) * m_width + (col - 1)) * 4;
			m_pixelMap[row][col] = Pixel_assemblePixel4(picData[curPixIndex], picData[curPixIndex + 1], picData[curPixIndex + 2], picData[curPixIndex + 3]);
		}
	}
	m_SDFTex->freePixelDataInCPU(picData);

	// 2. 计算图形边界
	for (int32 row = 1; row <= m_height; ++row)
	{
		for (int32 col = 1; col <= m_width; ++col)
		{
			// 当前像素不是完全透明并且与上下左右其中一个像素值不同
			if (m_pixelMap[row][col] != 0
				&& (m_pixelMap[row][col] != m_pixelMap[row - 1][col]
				|| m_pixelMap[row][col] != m_pixelMap[row + 1][col]
				|| m_pixelMap[row][col] != m_pixelMap[row][col - 1]
				|| m_pixelMap[row][col] != m_pixelMap[row][col + 1]))
			{
				// 更新 m_distanceMap 和 m_positionMap
				m_distanceMap[row][col] = 0;
				m_positionMap[row][col] = { int16(col), int16(row) };
			}
		}
	}

	// 3. 计算每个像素到轮廓的最近距离
	// 3.1 正向遍历
	for (int32 row = 1; row <= m_height; ++row)
	{
		for (int32 col = 1; col <= m_width; ++col)
		{
			float curr_d = m_distanceMap[row][col];
			Point16 curr_p = m_positionMap[row][col];
			Vector2 curPixel = vec2(col, row);
			// 检查正左方像素
			if (m_distanceMap[row][col - 1] + 1 < curr_d)
			{
				curr_p = m_positionMap[row][col - 1];
				Vector2 edgePixel = vec2(curr_p.x, curr_p.y);
				curr_d = (curPixel - edgePixel).length();
			}
			// 检查左上方像素
			if (m_distanceMap[row - 1][col - 1] + 1.414 < curr_d)
			{
				curr_p = m_positionMap[row - 1][col - 1];
				Vector2 edgePixel = vec2(curr_p.x, curr_p.y);
				curr_d = (curPixel - edgePixel).length();
			}
			// 检查正上方像素
			if (m_distanceMap[row - 1][col] + 1 < curr_d)
			{
				curr_p = m_positionMap[row - 1][col];
				Vector2 edgePixel = vec2(curr_p.x, curr_p.y);
				curr_d = (curPixel - edgePixel).length();
			}
			// 检查右上方像素
			if (m_distanceMap[row - 1][col + 1] + 1.414 < curr_d)
			{
				curr_p = m_positionMap[row - 1][col + 1];
				Vector2 edgePixel = vec2(curr_p.x, curr_p.y);
				curr_d = (curPixel - edgePixel).length();
			}
			m_distanceMap[row][col] = curr_d;
			m_positionMap[row][col] = curr_p;
		}
	}
	// 3.2 反向遍历
	for (int32 row = m_height; row >= 1; --row)
	{
		for (int32 col = m_width; col >= 1; --col)
		{
			float curr_d = m_distanceMap[row][col];
			Point16 curr_p = m_positionMap[row][col];
			Vector2 curPixel = vec2(col, row);
			// 检查正右方像素
			if (m_distanceMap[row][col + 1] + 1 < curr_d)
			{
				curr_p = m_positionMap[row][col + 1];
				Vector2 edgePixel = vec2(curr_p.x, curr_p.y);
				curr_d = (curPixel - edgePixel).length();
			}
			// 检查右下方像素
			if (m_distanceMap[row + 1][col + 1] + 1.414 < curr_d)
			{
				curr_p = m_positionMap[row + 1][col + 1];
				Vector2 edgePixel = vec2(curr_p.x, curr_p.y);
				curr_d = (curPixel - edgePixel).length();
			}
			// 检查正下方像素
			if (m_distanceMap[row + 1][col] + 1 < curr_d)
			{
				curr_p = m_positionMap[row + 1][col];
				Vector2 edgePixel = vec2(curr_p.x, curr_p.y);
				curr_d = (curPixel - edgePixel).length();
			}
			// 检查左下方像素
			if (m_distanceMap[row + 1][col - 1] + 1.414 < curr_d)
			{
				curr_p = m_positionMap[row + 1][col - 1];
				Vector2 edgePixel = vec2(curr_p.x, curr_p.y);
				curr_d = (curPixel - edgePixel).length();
			}
			m_distanceMap[row][col] = curr_d;
			m_positionMap[row][col] = curr_p;
		}
	}
	// 3.3 将图形内部的距离改为负值
	for (int32 row = 1; row <= m_height; ++row)
	{
		for (int32 col = 1; col <= m_width; ++col)
		{
			if (m_pixelMap[row][col] != 0)
			{
				m_distanceMap[row][col] *= -1;
			}
		}
	}

	// 4 根据多边形内部像素到最近边界的距离，修改像素的透明度
	for (int32 row = 1; row <= m_height; ++row)
	{
		for (int32 col = 1; col <= m_width; ++col)
		{
			float dist = m_distanceMap[row][col];
			if (dist < 0)
			{
				PixelType curColor = m_pixelMap[row][col];
				uint32 srcAlpha = Pixel_getA(curColor);
				float absDist = fabs(dist);
				float alpha = srcAlpha * (1.f - absDist / m_pixelDist);
				alpha = alpha < 0 ? 0 : alpha; 
				alpha = alpha > 255 ? 255 : alpha;
				PixelType afterColor = Pixel_combineAlpha(curColor, int(alpha));
				
				m_pixelMap[row][col] = afterColor;
			}
		}
	}

	// 5 更新 Texture 
	PixelType* updateData = new PixelType[m_width * m_height];
	for (int32 row = 1; row <= m_height; ++row)
	{
		for (int32 col = 1; col <= m_width; ++col)
		{
			int32 index = (row - 1) * m_width + col - 1;
			updateData[index] = m_pixelMap[row][col];
		}
	}
	Rect area = { 0, 0, m_width, m_height };
	m_SDFTex->update(&area, updateData);
	delete[] updateData;
}
