数据结构与算法——旋转数组的最小数字

December 12, 2024 · 1 min · Rick Cui

问题链接

要求： 空间复杂度：O(1) ，时间复杂度：O(logn)

思路： 分治

方法一：二分查找，如果左半边没有找到 nums[i] > nums[i+1] 的逆序对，则查找右半边的部分，如果都没找到，说明序列是有序的，直接返回索引 0 即可，空间复杂度：O(logn) ，时间复杂度：O(logn)

方法二：二分查找，如果 nums[mid] > nums[high] 则向右半边查找，如果 nums[mid] < nums[high] 则在左半边查找，如果相等则 high - 1 逐个判断

class Solution {
    // 方案一
    int calRotateIndex(vector<int>& nums, int l, int h) {
        if(l >= h) return -1;
        int mid = (l + h) / 2;
        if (nums[mid] > nums[mid+1]) {
            return mid+1;
        }
        int res = calRotateIndex(nums, l, mid);
        if (res == -1) {
            res = calRotateIndex(nums, mid + 1, h);
        }
        return res;
    }
    // 方案二
    int calRotateIndex2(vector<int>& nums, int l, int h) {
        while (l < h) {
            int mid = (l + h) / 2;
            if (nums[mid] < nums[h]) {
                h = mid;
            } else if (nums[mid] > nums[h]) {
                l = mid + 1;
            } else {
                h -= 1;
            }
        }
        return l;
    }
  public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     *
     * @param nums int整型vector
     * @return int整型
     */
    int minNumberInRotateArray(vector<int>& nums) {
        // write code here

        int l = 0, h = nums.size() - 1;
        
        int res = calRotateIndex(nums, l, h);
        if(res == -1){
            return nums[0];
        }
        return nums[res];
    }
};
