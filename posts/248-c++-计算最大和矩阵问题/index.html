C++——计算最大和子矩阵问题

September 27, 2022 · 3 min · Rick Cui

描述：计算矩阵中，最大和的子矩阵，并输出最大的和

测试用例一：最大和为 20， 最大子矩阵为后三列

$$\begin{equation} \begin{bmatrix*}[c] -3 & 5 & -1 & 5 \newline 2 & 4 & -2 & 4 \newline -1 & 3 & -1 & 3 \end{bmatrix*} \end{equation}$$

测试用例二：最大和为 12， 最大子矩阵为最后一列

$$\begin{equation} \begin{bmatrix*}[c] -3 & 2 & -1 & 5 \newline 5 & 4 & -2 & 4 \newline -1 & 3 & -1 & 3 \end{bmatrix*} \end{equation}$$

测试用例三：

$$\begin{equation} \begin{bmatrix*}[c] 1 & 2 & 3 & 4 \newline 5 & 6 & 7 & 8 \newline 9 & 10 & 11 & 12 \end{bmatrix*} \end{equation}$$

测试用例四：

$$\begin{equation} \begin{bmatrix*}[c] -1 & -2 & -3 & -4 \newline -5 & -6 & -7 & -8 \newline -9 & -10 & -11 & -12 \end{bmatrix*} \end{equation}$$

int main()
{
    // 测试用例一：
	/*int arr[3][4] = {
		{-3, 5, -1, 5},
		{2, 4, -2, 4},
		{-1, 3, -1, 3}
	};
	int dp[4][5]{ 0 };
	for (int i = 1; i < 4; ++i)
	{
		for (int j = 1; j < 5; ++j)
		{
			dp[i][j] = arr[i - 1][j - 1];
		}
	}*/
    // 测试用例二：
	/*int arr[3][4] = {
		{-3, 4, -1, 5},
		{5, 4, -2, 4},
		{-1, 3, -1, 3}
	};
	int dp[4][5]{ 0 };
	for (int i = 1; i < 4; ++i)
	{
		for (int j = 1; j < 5; ++j)
		{
			dp[i][j] = arr[i - 1][j - 1];
		}
	}*/
	int arr[3][4]{0}, dp[4][5]{0};
	int n = 1;
    // 测试用例三：
	/*for (int i = 0; i < 3; ++i)
	{
		for (int j = 0; j < 4; ++j) {
			arr[i][j] = n++;
			dp[i+1][j+1] = arr[i][j];
		}
	}*/
    // 测试用例四：
	for (int i = 0; i < 3; ++i)
	{
		for (int j = 0; j < 4; ++j) {
			arr[i][j] = -n++;
			dp[i+1][j+1] = arr[i][j];
		}
	}
    
	int cb = 0, ce = 0, rb = 0, re = 0;
	int max = arr[0][0];
	for (int i = 1; i < 4; ++i)
	{
		for (int j = 1; j < 5; ++j)
		{
			dp[i][j] += dp[i - 1][j] + dp[i][j - 1] - dp[i - 1][j - 1];
			if (max < dp[i][j] && dp[i][j] < arr[i - 1][j - 1])
			{
				max = arr[i - 1][j - 1];
				cb = ce = j - 1;
				rb = re = i - 1;
			}
			if (dp[i][j] > max)
			{
				max = dp[i][j];
				re = i-1;
				ce = j-1;
			}

		}
	}
	int maxSum = 0;
	for (int i = rb; i <= re; ++i)
	{
		for (int j = cb; j <= ce; ++j)
		{
			maxSum += arr[i][j];
		}
	}

	cout << "maxSum = " << maxSum << endl;
	cout << "rb = " << rb << " cb = " << cb << endl;
	cout << "re = " << re << " ce = " << ce << endl;
	return 0;
}
