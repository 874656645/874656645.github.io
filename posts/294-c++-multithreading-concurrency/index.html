C++——Multithreading Concurrency

#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <set>
#include <queue>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <chrono>
#include <algorithm>

using namespace std;

queue<string> mqueue;
condition_variable mcond;
mutex mmutex;

void consumer() {
	while (true) {
		unique_lock<mutex> lck{ mmutex }; // acquire mmutex
		mcond.wait(lck); /* do nothing */ // release lck and wait;
		// re-acquire lck upon wakeup

		cout << "queue size: " << mqueue.size() << endl;
		auto m = mqueue.front(); // get the message
		mqueue.pop();
		cout << this_thread::get_id() << " pop: " << m << endl;
		lck.unlock(); // release lck
		// ... process m ...
		//cout << m << endl;
	}
}

void producer(){
	int x{ 0 };
	while (true) {
		// ... fill the message ...
		unique_lock<mutex> lck{ mmutex }; // protect operations
		string m = to_string(x);
		mqueue.push(m);
		x++;
		cout << this_thread::get_id() << " push into queue\n";

		//mcond.notify_one(); // notify

		lck.unlock();
        
		// this_thread::sleep_for(chrono::seconds{ 1 });
	} // release lock (at end of scope)
}

void threadTest() {
	thread tc{ consumer };
	thread tp{ producer };

	tc.join();
	tp.join();
}
int main()
try
{
	threadTest();

	return 0;
}
catch (const std::exception&)
{
	cout << "exception" << endl;
}
