数据结构与算法——括号生成

December 13, 2024 · 1 min · Rick Cui

问题链接

思路： 递归的使用
左边括号数量小于总数
右边括号数量比左边括号数量少

class Solution {
    int sn;
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param n int整型 
     * @return string字符串vector
     */
    vector<string> generateParenthesis(int n) {
        // write code here
        vector<string> res;
        string tmp;
        sn = n;
        genP(0, 0, tmp, res);
        return res;
    }

    void genP(int left, int right, string tmp, vector<string>& res){
        if (left == sn && right == sn) {
            res.push_back(tmp);
            return;
        }
        if (left < sn) {
            genP(left+1, right, tmp+'(', res);
        }
        if (right < left) {
            genP(left, right+1, tmp+')', res);
        }
    }
};
