数据结构与算法——单链表的排序

December 12, 2024 · 1 min · Rick Cui

问题链接

要求：时间复杂度 O(nlogn)

思路：分治，归并排序思想 快慢指针，计算链表的中间节点 或者将链表节点放入到数组中，使用 sort 算法，最后再组成新的链表

/**
 * struct ListNode {
 *  int val;
 *  struct ListNode *next;
 *  ListNode(int x) : val(x), next(nullptr) {}
 * };
 */
class Solution {

  public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     *
     * @param head ListNode类 the head node
     * @return ListNode类
     */
    ListNode* sortInList(ListNode* head) {
        // write code here
        if(head == nullptr || head->next == nullptr){
            return head;
        }
        // 注意：如果 f = head（链表中有奇数个元素）
        //      while(f && f->next && f->next->next) 
        //      如果 f = head->next（链表中有偶数个元素）
        //      while(f && f->next)
        /*
        ListNode* s = head, * f = head;             // 奇数个元素
        while(f && f->next && f->next->next){
            s = s->next;
            f = f->next->next;
        }
        */
        ListNode* s = head, * f = head->next;       // 偶数个元素
        while(f && f->next){
            s = s->next;
            f = f->next->next;
        }
        ListNode* mid = s->next;
        s->next = nullptr;
        ListNode* left = sortInList(head);
        ListNode* right = sortInList(mid);

        // 合并
        ListNode* newHead = new ListNode(0);
        ListNode* tmp = newHead;
        while(left && right){
            if(left->val < right->val){
                tmp->next = left;
                left = left->next;
            }else{
                tmp->next = right;
                right = right->next;
            }
            tmp = tmp->next;
        }
        tmp->next = left == nullptr ? right : left;
        ListNode* res = newHead->next;
        newHead->next = nullptr;
        delete newHead;
        return res;
    }
};
